Ejemplo 2: Writing text...

Su texto: {{message}}

Su flag: {{flag}}

Descripción

                
                  var app = angular.module('ExampleApp2', []);

                  app.controller('mainController', function($scope) {
                    $scope.message = '';

                    $scope.cleanMessage = function() {
                      $scope.message = '';
                      $scope.flag = undefined;
                    }

                    $scope.trimMessage = function() {
                      $scope.message.trim();
                    }
                  });

En la plantilla (vista), insertamos: 

                
                <label for="writter-box">Escriba: </label>
                <input name="writter-box" type="text" ng-model="message" placeholder="Texto aquí..." />
                <p>Su texto: { { message } }</p>
                <p>Su flag: { { flag } }</p>
