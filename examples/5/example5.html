Ejemplo 5: Directiva molona

Lenguajes de programación

Descripción

Utilizamos una directiva personalizada para extender la funcionalidad HTML por defecto. Mediante esta etiqueta encapsularemos las siguientes características:

  • Un botón para eliminar el último elemento de la lista.
  • Mostrar la colección (array).
  • Una etiqueta que muestra en todo momento el número de elementos del array.

              
                var app = angular.module('ExampleApp1', []);
                app
                  .controller('ctrl1', function($scope) { 
                    $scope.languages = [
                      { name: 'Java', layer : 'BE' },
                      { name: 'Go', layer : 'BE' },
                      { name: 'JavaScript', layer : 'FE' },
                      { name: 'Scala', layer : 'BE' },
                      { name: 'Python', layer : 'BE' },
                      { name: 'Clojure', layer : 'BE' },
                      { name: 'Ruby', layer : 'BE' }
                    ];
                  })
                  .directive('semaDirective', function() {
                    return {
                      restrict : 'E',
                      scope : { smLangs : '=' },
                      template : '<button ng-click="remove()">Quitar último</button>' +
                                  '<div><ul>' +
                                  '<li ng-repeat="lang in smLangs">{{lang.name}}</li>' + 
                                  '</ul></div>' +
                                  '<br />' +
                                  '<p>Número de lenguajes: {{ smLangs.length }}</p>',
                      controller : function($scope) {
                        $scope.remove = function() {
                          $scope.smLangs.pop();
                        }
                      },
                      link : function(scope, element, attrs) {}
                    };
                  });

En la plantilla (vista), insertamos: 

              
                <h3>Lenguajes de programación</h3>
                <sema-directive sm-langs="languages"></sema-directive>
